    <script>
        // Use IndexedDB to save actual heavy video files in the browser locally
        let db;
        const request = indexedDB.open("MovieAppDB", 1);

        request.onupgradeneeded = function(e) {
            db = e.target.result;
            db.createObjectStore("movies", { keyPath: "id", autoIncrement: true });
        };

        request.onsuccess = function(e) {
            db = e.target.result;
            displayMovies('all');
        };

        // Modal open/close controls
        function toggleModal(show) {
            document.getElementById('uploadModal').style.display = show ? 'flex' : 'none';
        }

        // Form Submit Handler
        document.getElementById('uploadForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const title = document.getElementById('movieTitle').value;
            const genre = document.getElementById('movieGenre').value;
            const file = document.getElementById('movieFile').files[0];

            if (!file) return;

            const transaction = db.transaction(["movies"], "readwrite");
            const store = transaction.objectStore("movies");

            const newMovie = {
                title: title,
                genre: genre,
                videoBlob: file // Saving the actual file directly into DB
            };

            store.add(newMovie);

            transaction.oncomplete = function() {
                alert("Movie successfully added!");
                document.getElementById('uploadForm').reset();
                toggleModal(false);
                displayMovies('all');
            };
        });

        // Fetch and Render Movies from Database
        function displayMovies(filter = 'all') {
            const grid = document.getElementById('videoGrid');
            grid.innerHTML = ''; // Clear current display

            const transaction = db.transaction(["movies"], "readonly");
            const store = transaction.objectStore("movies");
            const getAllRequest = store.getAll();

            getAllRequest.onsuccess = function() {
                const movies = getAllRequest.result;
                const filteredMovies = filter === 'all' ? movies : movies.filter(m => m.genre === filter);

                if (filteredMovies.length === 0) {
                    grid.innerHTML = `<div class="no-movies">No movies found in this category. Be the first creator to upload!</div>`;
                    return;
                }

                filteredMovies.forEach(movie => {
                    // Create object URL from stored Blob video data
                    const videoURL = URL.createObjectURL(movie.videoBlob);

                    const card = document.createElement('div');
                    card.className = 'video-card';
                    card.innerHTML = `
                        <video src="${videoURL}" controls preload="metadata"></video>
                        <div class="video-info">
                            <div class="video-title">${movie.title}</div>
                            <span class="video-genre">${movie.genre}</span>
                        </div>
                    `;
                    grid.appendChild(card);
                });
            };
        }

        // Active Genre Toggle Buttons UI
        function filterGenre(genre) {
            const buttons = document.querySelectorAll('.genre-btn');
            buttons.forEach(btn => btn.classList.remove('active'));
            
            event.target.classList.add('active');
            displayMovies(genre);
        }
    </script>
